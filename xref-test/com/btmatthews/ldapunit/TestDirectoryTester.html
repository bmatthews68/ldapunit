17  package com.btmatthews.ldapunit;
18  
19  import org.junit.After;
20  import org.junit.Before;
21  import org.junit.Rule;
22  import org.junit.Test;
23  
24  import static org.junit.Assert.assertFalse;
25  import static org.junit.Assert.assertTrue;
26  
27  /**
28   * Unit test {@link DirectoryTester}.
29   *
30   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
31   * @since 1.0.0
32   */
33  @DirectoryServerConfiguration
34  public class TestDirectoryTester {
35  
36      /**
37       * This rule launches and shuts down the in-memory LDAP directory server.
38       */
39      @Rule
40      public DirectoryServerRule directoryServerRule = new DirectoryServerRule();
41      /**
42       * The {@link DirectoryTester} being tested.
43       */
44      private DirectoryTester tester;
45  
46      /**
47       * Prepare for test case execution by connecting to the LDAP directory.
48       */
49      @Before
50      public void setUp() {
51          tester = new DirectoryTester("localhost", 10389, "uid=admin,ou=system", "secret");
52      }
53  
54      /**
55       * Clean up after executing the test case by disconnecting from the LDAP directory server.
56       */
57      @After
58      public void tearDown() {
59          tester.disconnect();
60      }
61  
62      /**
63       * Verify should return true if the DN exists and false if it does not.
64       */
65      @Test
66      public void checkVerifyDNExists() {
67          assertTrue(tester.verifyDNExists("dc=btmatthews,dc=com"));
68          assertFalse(tester.verifyDNExists("ou=People,dc=btmatthews,dc=com"));
69      }
70  
71      /**
72       * Verify that the assertion succeeds when the DN does exist.
73       */
74      @Test
75      public void checkAssertDNExistsSucceeds() {
76          tester.assertDNExists("dc=btmatthews,dc=com");
77      }
78  
79      /**
80       * Verify that the assertion fails if the DN does not exist.
81       */
82      @Test(expected = AssertionError.class)
83      public void checkAssertDNExistsFails() {
84          tester.assertDNExists("ou=People,dc=btmatthews,dc=com");
85      }
86  
87      /**
88       * An exception is thrown when we try to verify an invalid DN.
89       */
90      @Test(expected = DirectoryTesterException.class)
91      public void throwsExceptionIfInvalidDN() {
92          tester.verifyDNExists("dc:btmatthews,dc:com");
93      }
94  
95      /**
96       * Verify that the {@link DirectoryTester#verifyDNHasAttribute(String, String)} behaves correctly.
97       */
98      @Test
99      @DirectoryServerConfiguration(ldifFiles = "com/btmatthews/ldapunit/initial.ldif")
100     public void checkVerifyDNHasAttribute() {
101         assertTrue(tester.verifyDNHasAttribute("dc=btmatthews,dc=com", "dc"));
102         assertFalse(tester.verifyDNHasAttribute("dc=btmatthews,dc=com", "ou"));
103         assertFalse(tester.verifyDNHasAttribute("ou=People,dc=btmatthews,dc=com", "dc"));
104         assertTrue(tester.verifyDNHasAttribute("ou=People,dc=btmatthews,dc=com", "ou"));
105         assertFalse(tester.verifyDNHasAttribute("ou=Groups,dc=btmatthews,dc=com", "ou"));
106     }
107 
108     /**
109      * Verify that the {@link DirectoryTester#verifyDNHasAttributeValue(String, String, String...)} behaves correctly.
110      */
111     @Test
112     @DirectoryServerConfiguration(ldifFiles = "com/btmatthews/ldapunit/initial.ldif")
113     public void checkVerifyDNIsA() {
114         assertTrue(tester.verifyDNIsA("dc=btmatthews,dc=com", "top"));
115         assertTrue(tester.verifyDNIsA("dc=btmatthews,dc=com", "domain"));
116         assertFalse(tester.verifyDNIsA("dc=btmatthews,dc=com", "organizationalUnit"));
117         assertTrue(tester.verifyDNIsA("ou=People,dc=btmatthews,dc=com", "top"));
118         assertTrue(tester.verifyDNIsA("ou=People,dc=btmatthews,dc=com", "organizationalUnit"));
119         assertFalse(tester.verifyDNIsA("ou=People,dc=btmatthews,dc=com", "inetOrgPerson"));
120         assertFalse(tester.verifyDNIsA("ou=Groips,dc=btmatthews,dc=com", "top"));
121     }
122 
123     @Test
124     @DirectoryServerConfiguration(ldifFiles = "com/btmatthews/ldapunit/initial.ldif")
125     public void checkVerifyDNHasAttributeValue() {
126         assertTrue(tester.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "objectclass", "top", "domain"));
127         assertTrue(tester.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "objectclass", "domain", "top"));
128         assertTrue(tester.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "btmatthews"));
129         assertFalse(tester.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "com"));
130         assertFalse(tester.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "btmatthews", "com"));
131         assertFalse(tester.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "ou", "People"));
132     }
133 
134     /**
135      * Verify that the {@link DirectoryTester#assertDNIsA(String, String)} method succeeds if the LDAP
136      * directory entry is a member of the object class.
137      */
138     @Test
139     public void assertDNIsAShouldSucceed() {
140         tester.assertDNIsA("dc=btmatthews,dc=com", "domain");
141     }
142 
143     /**
144      * Verify that the {@link DirectoryTester#assertDNIsA(String, String)} method throws an exception if the LDAP
145      * directory entry is not a member of the object class.
146      */
147     @Test(expected = AssertionError.class)
148     public void assertDNIsAShouldFail() {
149         tester.assertDNIsA("dc=btmatthews,dc=com", "organizationalUnit");
150     }
151 
152     /**
153      * Verify that the {@link DirectoryTester#assertDNHasAttribute(String, String)} method succeeds
154      * if the LDAP directory entry has the named attribute.
155      */
156     @Test
157     public void assertDNHasAttributeShouldSucceed() {
158         tester.assertDNHasAttribute("dc=btmatthews,dc=com", "dc");
159     }
160 
161     /**
162      * Verify that the {@link DirectoryTester#assertDNHasAttribute(String, String)} method throws an exception
163      * if the LDAP directory entry does not have the named attribute.
164      */
165     @Test(expected = AssertionError.class)
166     public void assertDNHasAttributeShouldFail() {
167         tester.assertDNHasAttribute("dc=btmatthews,dc=com", "ou");
168     }
169 
170     /**
171      * Verify that the {@link DirectoryTester#assertDNHasAttributeValue(String, String, String...)} method
172      * succeeds if the LDAP directory entry has a matching name/value pair.
173      */
174     @Test
175     public void assertDNHasAttributeValueShouldSucceed() {
176         tester.assertDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "btmatthews");
177     }
178 
179     /**
180      * Verify that the {@link DirectoryTester#assertDNHasAttributeValue(String, String, String...)} method
181      * throws an exception if the LDAP directory entry does not have the matching name/value pair.
182      */
183     @Test(expected = AssertionError.class)
184     public void assertDNHasAttributeValueShouldFail() {
185         tester.assertDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "com");
186     }
187 }
