40      /**
41       * Verify that the {@link DirectoryServerRule} launched an embedded directory server.
42       */
43      @Test
44      public void checkServerIsRunning() {
45          final DirectoryTester tester = new DirectoryTester("localhost", 10389, "uid=admin,ou=system", "secret");
46          try {
47              tester.assertDNExists("dc=btmatthews,dc=com");
48          } finally {
49              tester.disconnect();
50          }
51      }

55       */
56      @Test
57      public void checkVerifyDNExists() {
58          assertTrue(directoryServerRule.verifyDNExists("dc=btmatthews,dc=com"));
59          assertFalse(directoryServerRule.verifyDNExists("ou=People,dc=btmatthews,dc=com"));
60      }

86      /**
87       * Verify that the {@link DirectoryTester#verifyDNHasAttribute(String, String)} behaves correctly.
88       */
89      @Test
90      @DirectoryServerConfiguration(ldifFile = "com/btmatthews/ldapunit/initial.ldif")
91      public void checkVerifyDNHasAttribute() {
92          assertTrue(directoryServerRule.verifyDNHasAttribute("dc=btmatthews,dc=com", "dc"));
93          assertFalse(directoryServerRule.verifyDNHasAttribute("dc=btmatthews,dc=com", "ou"));
94          assertFalse(directoryServerRule.verifyDNHasAttribute("ou=People,dc=btmatthews,dc=com", "dc"));
95          assertTrue(directoryServerRule.verifyDNHasAttribute("ou=People,dc=btmatthews,dc=com", "ou"));
96          assertFalse(directoryServerRule.verifyDNHasAttribute("ou=Groups,dc=btmatthews,dc=com", "ou"));
97      }

99      /**
100      * Verify that the {@link DirectoryTester#verifyDNHasAttributeValue(String, String, String...)} behaves correctly.
101      */
102     @Test
103     @DirectoryServerConfiguration(ldifFile = "com/btmatthews/ldapunit/initial.ldif")
104     public void checkVerifyDNIsA() {
105         assertTrue(directoryServerRule.verifyDNIsA("dc=btmatthews,dc=com", "top"));
106         assertTrue(directoryServerRule.verifyDNIsA("dc=btmatthews,dc=com", "domain"));
107         assertFalse(directoryServerRule.verifyDNIsA("dc=btmatthews,dc=com", "organizationalUnit"));
108         assertTrue(directoryServerRule.verifyDNIsA("ou=People,dc=btmatthews,dc=com", "top"));
109         assertTrue(directoryServerRule.verifyDNIsA("ou=People,dc=btmatthews,dc=com", "organizationalUnit"));
110         assertFalse(directoryServerRule.verifyDNIsA("ou=People,dc=btmatthews,dc=com", "inetOrgPerson"));
111         assertFalse(directoryServerRule.verifyDNIsA("ou=Groips,dc=btmatthews,dc=com", "top"));
112     }
113 
114     @Test
115     @DirectoryServerConfiguration(ldifFile = "com/btmatthews/ldapunit/initial.ldif")
116     public void checkVerifyDNHasAttributeValue() {
117         assertTrue(directoryServerRule.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "objectclass", "top", "domain"));
118         assertTrue(directoryServerRule.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "objectclass", "domain", "top"));
119         assertTrue(directoryServerRule.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "btmatthews"));
120         assertFalse(directoryServerRule.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "com"));
121         assertFalse(directoryServerRule.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "dc", "btmatthews", "com"));
122         assertFalse(directoryServerRule.verifyDNHasAttributeValue("dc=btmatthews,dc=com", "ou", "People"));
123     }
