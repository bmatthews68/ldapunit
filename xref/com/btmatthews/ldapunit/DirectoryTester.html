17  package com.btmatthews.ldapunit;
18  
19  import com.unboundid.ldap.sdk.LDAPConnection;
20  import com.unboundid.ldap.sdk.LDAPException;
21  import com.unboundid.ldap.sdk.SearchResultEntry;
22  
23  /**
24   * Utility that maintains a connection to the LDAP directory server and provides assert and verify methods to
25   * test the LDAP directory contents.
26   *
27   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
28   * @since 1.0.0
29   */
30  public class DirectoryTester {
31  
32      /**
33       * The connection to the LDAP directory server.
34       */
35      private final LDAPConnection connection;
36  
37      /**
38       * Initialise the LDAP directory tester by connecting to the LDAP directory server using the {@code hostname} and
39       * {@code port}.
40       *
41       * @param hostname The host name of the directory server.
42       * @param port     The TCP port number of the directory server.
43       * @throws DirectoryTesterException If there was a problem connecting to the LDAP directory server.
44       */
45      public DirectoryTester(final String hostname, final int port) {
46          try {
47              connection = new LDAPConnection(hostname, port);
48          } catch (final LDAPException e) {
49              throw new DirectoryTesterException("Could not connect to LDAP directory server", e);
50          }
51      }
52  
53      /**
54       * Initialise the LDAP directory tester by connecting to the LDAP directory server using the {@code hostname} and
55       * {@code port} and bind to it using the {@code bindDN} and {@code password}.
56       *
57       * @param hostname The host name of the directory server.
58       * @param port     The TCP port number of the directory server.
59       * @param bindDN   The DN used to bind to the LDAP directory server.
60       * @param password The password used to bind to the LDAP directory server.
61       * @throws DirectoryTesterException If there was a problem connecting to the LDAP directory server.
62       */
63      public DirectoryTester(final String hostname, final int port, final String bindDN, final String password) {
64          this(hostname, port);
65          try {
66              connection.bind(bindDN, password);
67          } catch (final LDAPException e) {
68              throw new DirectoryTesterException("Could not bind to LDAP directory server", e);
69          }
70      }
71  
72      /**
73       * Verify that an entry identified by {@code dn} exists.
74       *
75       * @param dn The distinguished name.
76       * @return {@code true} if an entry identified by {@code dn} exists. Otherwise, {@code false} is returned.
77       */
78      public boolean verifyDNExists(final String dn) {
79          try {
80              final SearchResultEntry entry = connection.getEntry(dn);
81              return entry != null;
82          } catch (final LDAPException e) {
83              throw new DirectoryTesterException("Error communicating with LDAP directory server", e);
84          }
85      }
86  
87      /**
88       * Assert that an entry identified by {@code dn} exists.
89       *
90       * @param dn The distinguished name.
91       */
92      public void assertDNExists(final String dn) {
93          if (!verifyDNExists(dn)) {
94              final StringBuilder message = new StringBuilder("DN does not exist: ");
95              message.append(dn);
96              throw new AssertionError(message);
97          }
98      }
99  
100     /**
101      * Disconnect from the LDAP directory server.
102      */
103     public void disconnect() {
104         connection.close();
105     }
106 }
