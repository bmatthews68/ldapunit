32  /**
33   * Helper functions to start the in-memory LDAP directory server, load LDAP directory entries from an LDIF files and
34   * shut down the in-memory LDAP directory server.
35   *
36   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
37   * @since 1.0.1
38   */
39  public final class DirectoryServerUtils {
40  
41      /**
42       * Create and configure an embedded LDAP directory server, load seed data and start the server.
43       *
44       * @param ldifFile The LDIF resource or file from which LDIF records will be loaded.
45       * @return The {@link com.unboundid.ldap.listener.InMemoryDirectoryServer} object.
46       * @throws com.unboundid.ldif.LDIFException     If there was an error in the LDIF data.
47       * @throws com.unboundid.ldap.sdk.LDAPException If there was a problem configuring or starting the embedded LDAP directory server.
48       * @throws java.io.IOException                  If there was a problem reading the LDIF data.
49       */
50      @Deprecated
51      public static InMemoryDirectoryServer startServer(final int port,
52                                                        final String baseDN,
53                                                        final String authDN,
54                                                        final String authPassword,
55                                                        final String ldifFile)
56              throws LDIFException, LDAPException, IOException {
57          return startServer(port, baseDN, authDN, authPassword, new String[]{ldifFile});
58      }
59  
60      /**
61       * Create and configure an embedded LDAP directory server, load seed data and start the server.
62       *
63       * @param ldifFiles The LDIF resources or files from which LDIF records will be loaded.
64       * @return The {@link com.unboundid.ldap.listener.InMemoryDirectoryServer} object.
65       * @throws com.unboundid.ldif.LDIFException     If there was an error in the LDIF data.
66       * @throws com.unboundid.ldap.sdk.LDAPException If there was a problem configuring or starting the embedded LDAP directory server.
67       * @throws java.io.IOException                  If there was a problem reading the LDIF data.
68       */
69      public static InMemoryDirectoryServer startServer(final int port,
70                                                        final String baseDN,
71                                                        final String authDN,
72                                                        final String authPassword,
73                                                        final String[] ldifFiles)
74              throws LDIFException, LDAPException, IOException {
75          final InMemoryListenerConfig listenerConfig = InMemoryListenerConfig.createLDAPConfig("default", port);
76          final InMemoryDirectoryServerConfig config = new InMemoryDirectoryServerConfig(new DN(baseDN));
77          config.setListenerConfigs(listenerConfig);
78          config.addAdditionalBindCredentials(authDN, authPassword);
79          final InMemoryDirectoryServer server = new InMemoryDirectoryServer(config);
80          server.add(new Entry(baseDN, new Attribute("objectclass", "domain", "top")));
81          server.startListening();
82          for (final String ldifFile : ldifFiles) {
83              loadData(server, ldifFile);
84          }
85          return server;
86      }
87  
88      /**
89       * Load LDIF records from a file to seed the LDAP directory.
90       *
91       * @param server   The embedded LDAP directory server.
92       * @param ldifFile The LDIF resource or file from which LDIF records will be loaded.
93       * @throws com.unboundid.ldif.LDIFException     If there was an error in the LDIF data.
94       * @throws com.unboundid.ldap.sdk.LDAPException If there was a problem loading the LDIF records into the LDAP directory.
95       * @throws java.io.IOException                  If there was a problem reading the LDIF records from the file.
96       */
97      public static void loadData(final InMemoryDirectoryServer server,
98                                  final String ldifFile)
99              throws LDIFException, LDAPException, IOException {
100         if (ldifFile != null && !ldifFile.isEmpty()) {
101             final InputStream inputStream;
102             final File file = new File(ldifFile);
103             if (file.exists()) {
104                 inputStream = new FileInputStream(file);
105             } else {
106                 final ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
107                 inputStream = classLoader.getResourceAsStream(ldifFile);
108             }
109             if (inputStream != null) {
110                 loadData(server.getConnection(), inputStream);
111             }
112         }
113     }
114 
115     private static void loadData(final LDAPConnection connection,
116                                  final InputStream inputStream)
117             throws LDIFException, LDAPException, IOException {
118         try {
119             final LDIFReader reader = new LDIFReader(inputStream);
120             LDIFChangeRecord changeRecord = reader.readChangeRecord(true);
121             while (changeRecord != null) {
122                 changeRecord.processChange(connection);
123                 changeRecord = reader.readChangeRecord(true);
124             }
125         } finally {
126             inputStream.close();
127         }
128     }
129 
130     /**
131      * Shutdown the embedded LDAP directory server.
132      *
133      * @param server The embedded LDAP directory server.
134      */
135     public static void stopServer(final InMemoryDirectoryServer server) {
136         server.shutDown(true);
137     }
138 }
