32  /**
33   * Helper functions to start the in-memory LDAP directory server, load LDAP directory entries from an LDIF files and
34   * shut down the in-memory LDAP directory server.
35   *
36   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
37   * @since 1.0.1
38   */
39  public final class DirectoryServerUtils {
40  
41      /**
42       * Create and configure an embedded LDAP directory server, load seed data and start the server.
43       *
44       * @param ldifFile The LDIF resource or file from which LDIF records will be loaded.
45       * @return The {@link com.unboundid.ldap.listener.InMemoryDirectoryServer} object.
46       * @throws com.unboundid.ldap.sdk.LDAPException
47       *                             If there was a problem configuring or starting the embedded LDAP directory server.
48       * @throws java.io.IOException If there was a problem reading the LDIF data.
49       */
50      public static InMemoryDirectoryServer startServer(final int port,
51                                                        final String baseDN,
52                                                        final String authDN,
53                                                        final String authPassword,
54                                                        final String ldifFile) throws LDAPException, IOException {
55          final InMemoryListenerConfig listenerConfig = InMemoryListenerConfig.createLDAPConfig("default", port);
56          final InMemoryDirectoryServerConfig config = new InMemoryDirectoryServerConfig(new DN(baseDN));
57          config.setListenerConfigs(listenerConfig);
58          config.addAdditionalBindCredentials(authDN, authPassword);
59          final InMemoryDirectoryServer server = new InMemoryDirectoryServer(config);
60          server.add(new Entry(baseDN, new Attribute("objectclass", "domain", "top")));
61          loadData(server, ldifFile);
62          server.startListening();
63          return server;
64      }
65  
66      /**
67       * Load LDIF records from a file to seed the LDAP directory.
68       *
69       * @param server   The embedded LDAP directory server.
70       * @param ldifFile The LDIF resource or file from which LDIF records will be loaded.
71       * @throws com.unboundid.ldap.sdk.LDAPException
72       *                             If there was a problem loading the LDIF records into the LDAP directory.
73       * @throws java.io.IOException If there was a problem reading the LDIF records from the file.
74       */
75      public static void loadData(final InMemoryDirectoryServer server,
76                                  final String ldifFile) throws LDAPException, IOException {
77          if (ldifFile != null && !ldifFile.isEmpty()) {
78              if (new File(ldifFile).exists()) {
79                  server.importFromLDIF(false, ldifFile);
80              } else {
81                  final ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
82                  final InputStream inputStream = classLoader.getResourceAsStream(ldifFile);
83                  try {
84                      final LDIFReader reader = new LDIFReader(inputStream);
85                      server.importFromLDIF(false, reader);
86                  } finally {
87                      inputStream.close();
88                  }
89              }
90          }
91      }
92  
93      /**
94       * Shutdown the embedded LDAP directory server.
95       *
96       * @param server The embedded LDAP directory server.
97       */
98      public static void stopServer(final InMemoryDirectoryServer server) {
99          server.shutDown(true);
100     }
