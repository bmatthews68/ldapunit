17  package com.btmatthews.ldapunit;
18  
19  import org.junit.rules.TestRule;
20  import org.junit.runner.Description;
21  import org.junit.runners.model.Statement;
22  
23  /**
24   * A JUnit test rule that starts an embedded LDAP directory server. The configuration for the directory server
25   * is obtained from the {@link DirectoryServerConfiguration} annotation applied to either the test method or the test
26   * class.
27   *
28   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
29   * @since 1.0.0
30   */
31  public class DirectoryServerRule implements TestRule {
32  
33      /**
34       * Modifies the method-running {@link Statement} to implement this test-running rule. The configuration for
35       * the embedded LDAP directory server is obtained from the {@link DirectoryServerConfiguration} annotation that was
36       * applied to either the test class or test method.
37       *
38       * @param base        The {@link Statement} to be modified
39       * @param description A {@link Description} of the test implemented in {@code base}
40       * @return If no configuration was found then {@code base} is returned. Otherwise, a new
41       *         {@link DirectoryServerStatement} that wraps {@code base} is returned.
42       */
43      @Override
44      public Statement apply(final Statement base, final Description description) {
45          DirectoryServerConfiguration annotation = description.getAnnotation(DirectoryServerConfiguration.class);
46          if (annotation == null) {
47              final Class testClass = description.getTestClass();
48              annotation = (DirectoryServerConfiguration) testClass.getAnnotation(DirectoryServerConfiguration.class);
49          }
50          if (annotation != null) {
51              return new DirectoryServerStatement(base, annotation);
52          }
53          return base;
54      }
55  }
