29  import java.io.File;
30  import java.io.IOException;
31  import java.io.InputStream;

33  /**
34   * A {@link Statement} wrapper that launches an embedded LDAP directory server before executing the wrapped statement.
35   * The embedded LDAP directory server is shutdown after the wrapped statement has been executed.
36   *
37   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
38   * @since 1.0.0
39   */
40  public class DirectoryServerStatement extends Statement {
41  
42      /**
43       * The wrapped statement {@link Statement}.
44       */
45      private final Statement base;
46      /**
47       * The annotation that defines the directory server configuration.
48       */
49      private final DirectoryServerConfiguration annotation;
50  
51      /**
52       * Initialise the wrapper statement that starts an embedded LDAP directory server and shuts it down before and after
53       * executing the wrapped statement.
54       *
55       * @param stmt The wrapped statement.
56       * @param cfg  The directory server configuration.
57       */
58      public DirectoryServerStatement(final Statement stmt, final DirectoryServerConfiguration cfg) {
59          base = stmt;
60          annotation = cfg;
61      }
62  
63      /**
64       * Start an embedded LDAP directory server before executing the wrapped statement and shutdown the LDAP directory
65       * server after the wrapped statement completes.
66       *
67       * @throws Throwable If there was an error starting the server, executing the wrapped statement or shutting the
68       *                   wrapped server.
69       */
70      @Override
71      public void evaluate() throws Throwable {
72          final InMemoryDirectoryServer server = startServer();
73          try {
74              base.evaluate();
75          } finally {
76              stopServer(server);
77          }
78      }
79  
80      /**
81       * Create and configure an embedded LDAP directory server, load seed data and start the server.
82       *
83       * @return The {@link InMemoryDirectoryServer} object.
84       * @throws LDAPException If there was a problem configuring or starting the embedded LDAP directory server.
85       * @throws IOException   If there was a problem reading the LDIF data.
86       */
87      private InMemoryDirectoryServer startServer() throws LDAPException, IOException {
88          final InMemoryListenerConfig listenerConfig = InMemoryListenerConfig.createLDAPConfig("default", annotation.port());
89          final InMemoryDirectoryServerConfig config = new InMemoryDirectoryServerConfig(new DN(annotation.baseDN()));
90          config.setListenerConfigs(listenerConfig);
91          config.addAdditionalBindCredentials(annotation.authDN(), annotation.authPassword());
92          final InMemoryDirectoryServer server = new InMemoryDirectoryServer(config);
93          server.add(new Entry(annotation.baseDN(), new Attribute("objectclass", "domain", "top")));
94          loadData(server);
95          server.startListening();
96          return server;
97      }
98  
99      /**
100      * Load LDIF records from a file to seed the LDAP directory.
101      *
102      * @param server The embedded LDAP directory server.
103      * @throws LDAPException If there was a problem loading the LDIF records into the LDAP directory.
104      * @throws IOException   If there was a problem reading the LDIF records from the file.
105      */
106     private void loadData(final InMemoryDirectoryServer server) throws LDAPException, IOException {
107         final String ldifFile = annotation.ldifFile();
108         if (ldifFile != null && !ldifFile.isEmpty()) {
109             if (new File(ldifFile).exists()) {
110                 server.importFromLDIF(false, ldifFile);
111             } else {
112                 final ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
113                 final InputStream inputStream = classLoader.getResourceAsStream(ldifFile);
114                 try {
115                     final LDIFReader reader = new LDIFReader(inputStream);
116                     server.importFromLDIF(false, reader);
117                 } finally {
118                     inputStream.close();
119                 }
120             }
121         }
122     }
123 
124     /**
125      * Shutdown the embedded LDAP directory server.
126      *
127      * @param server The embedded LDAP directory server.
128      */
129     private void stopServer(final InMemoryDirectoryServer server) {
130         server.shutDown(true);
131     }
