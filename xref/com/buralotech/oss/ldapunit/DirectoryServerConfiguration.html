17  package com.buralotech.oss.ldapunit;
18  
19  import java.lang.annotation.ElementType;
20  import java.lang.annotation.Retention;
21  import java.lang.annotation.RetentionPolicy;
22  import java.lang.annotation.Target;
23  
24  /**
25   * This annotation is used to specify the parameters used to configure the LDAP directory server. The
26   * annotation can be applied to the test class or method. If the annotation has been applied to both
27   * the test class and the test method the the annotation on the test method is used.
28   *
29   * @author <a href="mailto:bmatthews68@gmail.com">Brian Matthews</a>
30   * @since 1.0.0
31   */
32  @Target({ElementType.TYPE, ElementType.METHOD})
33  @Retention(RetentionPolicy.RUNTIME)
34  public @interface DirectoryServerConfiguration {
35  
36      /**
37       * The default TCP port for the directory server.
38       */
39      int DEFAULT_PORT = 10389;
40  
41      /**
42       * The TCP port that the LDAP directory server will be configured to listen on.
43       *
44       * @return The TCP port.
45       */
46      int port() default DEFAULT_PORT;
47  
48      /**
49       * The DN that will be configured as the root of the LDAP directory.
50       *
51       * @return The DN.
52       */
53      String baseDN() default "dc=buralotech,dc=com";
54  
55      /**
56       * The object classes used for the base DN.
57       *
58       * @return The object classes.
59       */
60      String[] baseObjectClasses() default {"domain", "top"};
61  
62      /**
63       * The attribute name/value pairs used for the base DN.
64       *
65       * @return The attribute name/value pairs.
66       */
67      String[] baseAttributes() default {};
68  
69      /**
70       * The DN that will be configured as the administrator account identifier.
71       *
72       * @return The DN.
73       */
74      String authDN() default "uid=admin,ou=system";
75  
76      /**
77       * The password that will be configured as the authentication credentials for the administrator account.
78       *
79       * @return The password.
80       */
81      String authPassword() default "secret";
82  
83      /**
84       * The locations of optional LDIF files that can be used to see the LDAP directory with an initial data set.
85       * The files may localed on the file system or the classpath. The classpath is checked first and then falls back
86       * to che file system if it was not found on the class path.
87       *
88       * @return An array of LDIF file paths.
89       */
90      String[] ldifFiles() default {};
91  
92      /**
93       * The location of the optional schema LDIF files that can be use to have a LDAP directory with a custom schema.
94       * The file may located on the file system or the classpath. The classpath is checked first and then falls back
95       * to the file system if it was no found on the class path.
96       *
97       * @return The file paths of the schemas in LDIF.
98       */
99      String[] schemaFiles() default {};
100 }
