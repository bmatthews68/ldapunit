38  /**
39   * Helper functions to start the in-memory LDAP directory server, load LDAP directory entries from an LDIF files and
40   * shut down the in-memory LDAP directory server.

42   * @author <a href="mailto:bmatthews68@gmail.com">Brian Matthews</a>

48       * Hidden constructor.

53      /**
54       * Start the directory server using the configuration specified by the {@link DirectoryServerConfiguration}
55       * annotation.
56       *
57       * @param annotation The configuration.
58       * @return The {@link  InMemoryDirectoryServer} object.
59       * @throws LDIFException If there was an error in the LDIF data.
60       * @throws LDAPException If there was a problem configuring or starting the embedded LDAP directory server.
61       * @throws IOException   If there was a problem reading the LDIF data.
62       */
63      static InMemoryDirectoryServer startServer(final DirectoryServerConfiguration annotation)
64              throws LDIFException, LDAPException, IOException {
65          return startServer(
66                  annotation.port(),
67                  annotation.baseDN(),
68                  annotation.baseObjectClasses(),
69                  annotation.baseAttributes(),
70                  annotation.authDN(),
71                  annotation.authPassword(),
72                  annotation.ldifFiles(),
73                  annotation.schemaFiles());
74      }

76      /**
77       * Create and configure an embedded LDAP directory server, load seed data and start the server.
78       *
79       * @param port              The TCP port that the LDAP directory server will be configured to listen on.
80       * @param baseDN            The DN that will be configured as the root of the LDAP directory.
81       * @param baseObjectClasses The object classes to use when creating the base DN.
82       * @param baseAttributes    The attributes to set on the base DN.
83       * @param authDN            The DN that will be configured as the administrator account identifier.
84       * @param authPassword      The password that will be configured as the authentication credentials for
85       *                          the administrator account.
86       * @param ldifFiles         The LDIF resources or files from which LDIF records will be loaded.
87       * @param schemaFiles       The files from which to load custom schemas.
88       * @return The {@link  InMemoryDirectoryServer} object.
89       * @throws LDIFException If there was an error in the LDIF data.
90       * @throws LDAPException If there was a problem configuring or starting the embedded LDAP directory server.
91       * @throws IOException   If there was a problem reading the LDIF data.
92       */
93      static InMemoryDirectoryServer startServer(final int port,
94                                                 final String baseDN,
95                                                 final String[] baseObjectClasses,
96                                                 final String[] baseAttributes,
97                                                 final String authDN,
98                                                 final String authPassword,
99                                                 final String[] ldifFiles,
100                                                final String[] schemaFiles)
101             throws LDIFException, LDAPException, IOException {
102         final InMemoryListenerConfig listenerConfig = InMemoryListenerConfig.createLDAPConfig("default", port);
103         final InMemoryDirectoryServerConfig config = new InMemoryDirectoryServerConfig(new DN(baseDN));
104         loadSchema(config, schemaFiles);
105         config.setListenerConfigs(listenerConfig);
106         config.addAdditionalBindCredentials(authDN, authPassword);
107         final InMemoryDirectoryServer server = new InMemoryDirectoryServer(config);
108         final int n = baseAttributes.length;
109         final Attribute[] attributes = new Attribute[1 + n];
110         for (int i = 0; i < n; i++) {
111             final int j = baseAttributes[i].indexOf('=');
112             final String name;
113             final String value;
114             if (j == -1) {
115                 name = baseAttributes[i];
116                 value = "";
117             } else {
118                 name = baseAttributes[i].substring(0, j);
119                 value = baseAttributes[i].substring(j + 1);
120             }
121             attributes[i] = new Attribute(name, value);
122         }
123         attributes[n] = new Attribute("objectclass", baseObjectClasses);
124         server .add(new Entry(baseDN, attributes));
125         server.startListening();
126         for (final String ldifFile : ldifFiles) {
127             loadData(server, ldifFile);
128         }
129         return server;
130     }

132     /**
133      * Look for path on the classpath or file system and return an {@link InputStream} if found.
134      *
135      * @param path The path.
136      * @return An {@link InputStream} if the path exists on the classpath or file sytem. Otherwise, {@code null}.
137      */
138     private static InputStream getInputStream(final String path) {
139         if (path == null || path.isEmpty()) {
140             return null;
141         }
142         final ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
143         final InputStream inputStream = classLoader.getResourceAsStream(path);
144         if (inputStream == null) {
145             final File file = new File(path);
146             if (file.exists()) {
147                 try {
148                     return new FileInputStream(file);
149                 } catch (final FileNotFoundException e) {
150                     return null;
151                 }
152             } else {
153                 return null;
154             }
155         } else {
156             return inputStream;
157         }
158     }

160     /**
161      * Load a custom schema.
162      *
163      * @param config      The directory server configuration.
164      * @param schemaFiles The schema files.
165      * @throws LDIFException If there was a problem loading the schema into the LDAP directory.
166      * @throws LDAPException If there was a problem loading the schema into the LDAP directory.
167      * @throws IOException   If there was a problem reading the schema from the file.
168      */
169     private static void loadSchema(final InMemoryDirectoryServerConfig config,
170                                    final String[] schemaFiles)
171             throws LDIFException, LDAPException, IOException {
172         if (schemaFiles.length > 0) {
173             final Schema[] schemas = new Schema[schemaFiles.length];
174             for (int i = 0; i < schemaFiles.length; i++) {
175                 if ("default".equals(schemaFiles[i])) {
176                     schemas[i] = Schema.getDefaultStandardSchema();
177                 } else {
178                     try (InputStream inputStream = getInputStream(schemaFiles[i])) {
179                         if (inputStream != null) {
180                             schemas[i] = Schema.getSchema(inputStream);
181                         }
182                     }
183                 }
184             }
185             config.setSchema(Schema.mergeSchemas(schemas));
186         }
187     }

189     /**
190      * Load LDIF records from a file to seed the LDAP directory.
191      *
192      * @param server   The embedded LDAP directory server.
193      * @param ldifFile The LDIF resource or file from which LDIF records will be loaded.
194      * @throws LDIFException If there was an error in the LDIF data.
195      * @throws LDAPException If there was a problem loading the LDIF records into the LDAP directory.
196      * @throws IOException   If there was a problem reading the LDIF records from the file.
197      */
198     private static void loadData(final InMemoryDirectoryServer server,
199                                  final String ldifFile)
200             throws LDIFException, LDAPException, IOException {
201         try (InputStream inputStream = getInputStream(ldifFile)) {
202             if (inputStream != null) {
203                 loadData(server.getConnection(), inputStream);
204             }
205         }
206     }
207 
208     /**
209      * Load LDIF records from an input stream to seed the LDAP directory.
210      *
211      * @param connection  A connection to the LDAP directory.
212      * @param inputStream TThe input stream from which LDIF records will be loaded.
213      * @throws LDIFException If there was an error in the LDIF data.
214      * @throws LDAPException If there was a problem loading the LDIF records into the LDAP directory.
215      * @throws IOException   If there was a problem reading the LDIF records from the file.
216      */
217     private static void loadData(final LDAPConnection connection,
218                                  final InputStream inputStream)
219             throws LDIFException, LDAPException, IOException {
220         try (LDIFReader reader = new LDIFReader(inputStream)) {
221             LDIFChangeRecord changeRecord = reader.readChangeRecord(true);
222             while (changeRecord != null) {
223                 changeRecord.processChange(connection);
224                 changeRecord = reader.readChangeRecord(true);
225             }
226         }
227     }

229     /**
230      * Shutdown the embedded LDAP directory server.
231      *
232      * @param server The embedded LDAP directory server.
233      */
234     static void stopServer(final InMemoryDirectoryServer server) {
235         server.shutDown(true);
236     }
